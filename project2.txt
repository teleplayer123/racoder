  use anyhow::{Result, Context};
  use clap::Parser;
  use std::path::PathBuf;
  use tokio::sync::mpsc;
  use ratatui::prelude::*;

  #[derive(Parser)]
  #[command(author, version, about, long_about = None)]
  struct Args {
      /// URL to llama.cpp server
      #[arg(short, long, default_value = "http://localhost:8080")]
      server_url: String,

      /// Model name
      #[arg(short, long, default_value = "qwen3.5")]
      model: String,

      /// Temperature (0.0-2.0)
      #[arg(short, long, default_value = "0.7")]
      temperature: f32,

      /// System prompt for the agent
      #[arg(long, default_value = "You are a coding assistant. Help me with coding tasks.")]
      system_prompt: String,
  }

  async fn chat(messages: &mut Vec<tools::Message>, temperature: f32) -> Result<()> {
      let mut client = llama_client::Client::new();

      let stream = client
          .chat(messages, temperature)
          .await
          .context("Failed to get chat response")?;

      let mut response = String::new();

      while let Some(chunk) = stream
          .receive()
          .await
          .context("Failed to receive stream")?
      {
          let token = serde_json::from_str::<tools::Token>(&chunk)
              .ok()
              .context("Failed to parse chunk")?;

          match token {
              tools::Token::Choice(choice) => {
                  response.push_str(choice);
              }
              tools::Token::Error(err) => {
                  return Err(anyhow::anyhow!(err));
              }
          }
      }

      let output = response;

      // Save to history
      tools::add_to_history(&output)?;

      println!("{}", output);

      Ok(())
  }

  async fn run_app(args: &Args) -> Result<()> {
      // Load system prompt
      let system_prompt = args.system_prompt.clone();

      let mut messages = vec![tools::Message {
          role: "system".to_string(),
          content: system_prompt.clone(),
      }];

      // Main loop
      loop {
          // Read input
          print!("> ");
          io::stdout().flush()?;
          let input = io::stdin().read_line()?;

          if input.trim().is_empty() {
              continue;
          }

          if input.trim() == "exit" || input.trim() == "/quit" {
              break;
          }

          // Check for file commands
          if input.starts_with('/') {
              let command = input.trim()[1..].to_lowercase();

              match command.as_str() {
                  "read" => {
                      // Ask for filename
                      print!("Read: ");
                      io::stdout().flush()?;
                      let file_path = io::stdin().read_line()?;

                      let content = std::fs::read_to_string(&file_path)
                          .ok()
                          .unwrap_or_default();

                      if let Some(content) = content {
                          messages.push(tools::Message {
                              role: "user".to_string(),
                              content: format!("Read: {}\n{}", file_path, content),
                          });
                      }
                  }
                  "write" | "create" => {
                      // Ask for filename and content
                      print!("Write: ");
                      io::stdout().flush()?;
                      let file_path = io::stdin().read_line()?;

                      print!("Content: ");
                      io::stdout().flush()?;
                      let content = io::stdin().read_to_string()?;

                      if let Err(e) = std::fs::write(&file_path, content) {
                          eprintln!("Error writing file: {}", e);
                      }

                      messages.push(tools::Message {
                          role: "system".to_string(),
                          content: format!("I wrote to: {}\n{}", file_path, content),
                      });
                  }
                  "edit" => {
                      // Basic edit - replace text in file
                      let (path, old, new) = {
                          print!("Edit file: ");
                          io::stdout().flush()?;
                          let path = io::stdin().read_line()?;

                          print!("Old text: ");
                          io::stdout().flush()?;
                          let old = io::stdin().read_line()?;

                          print!("New text: ");
                          io::stdout().flush()?;
                          let new = io::stdin().read_line()?;

                          let mut content = std::fs::read_to_string(&path)?;
                          if let Some(pos) = content.find(old) {
                              content.replace_range(pos..pos + old.len(), new);
                          } else {
                              eprintln!("Text '{}' not found in file", old);
                          }

                          std::fs::write(&path, content)?;
                          (path, old, new)
                      };

                      messages.push(tools::Message {
                          role: "system".to_string(),
                          content: format!("Edited: {} (replaced '{}'' with '{}')",
                              path.display(), old, new),
                      });
                  }
                  _ => {
                      // Just regular chat
                      messages.push(tools::Message {
                          role: "user".to_string(),
                          content: input.clone(),
                      });
                  }
              }
          } else {
              // Regular chat message
              messages.push(tools::Message {
                  role: "user".to_string(),
                  content: input,
              });
          }

          // Send to LLM
          if messages.len() > 1 {
              messages.push(tools::Message {
                  role: "assistant".to_string(),
                  content: String::new(),
              });
              chat(&mut messages, args.temperature).await?;
          }
      }

      Ok(())
  }

  #[tokio::main]
  async fn main() -> Result<()> {
      let args = Args::parse();

      // Load history
      tools::load_history()?;

      // Clear terminal
      crossterm::terminal::enable_raw_mode()?;
      crossterm::execute!(io::stdout(), crossterm::terminal::Clear,
  crossterm::cursor::Hide)?;

      // Run app
      run_app(&args).await?;

      // Restore terminal
      crossterm::terminal::disable_raw_mode()?;
      crossterm::execute!(io::stdout(), crossterm::cursor::Show)?;

      Ok(())
  }

  use anyhow::{Context, Result};
  use reqwest::Client;
  use tokio::io::AsyncBufReadExt;
  use futures::StreamExt;

  #[derive(Debug, Clone)]
  pub struct Client {
      url: String,
      api_key: Option<String>,
  }

  #[derive(Debug, Clone)]
  pub struct ChatRequest {
      model: String,
      messages: Vec<Message>,
      temperature: f32,
  }

  #[derive(Debug, Clone, serde::Serialize)]
  pub struct Message {
      role: String,
      content: String,
  }

  #[derive(Debug, Clone, serde::Deserialize)]
  pub struct ChatResponse {
      id: String,
      object: String,
      created: u64,
      model: String,
      choices: Vec<Choice>,
  }

  #[derive(Debug, Clone, serde::Deserialize)]
  pub struct Choice {
      index: u32,
      message: Message,
      finish_reason: Option<String>,
  }

  #[derive(Debug, Clone, serde::Deserialize)]
  pub struct Message {
      role: String,
      content: String,
  }

  impl Client {
      pub fn new() -> Self {
          Client {
              url: String::from("http://localhost:8080/v1"),
              api_key: None,
          }
      }

      pub async fn chat(&self, messages: &Vec<Message>, temperature: f32) ->
  Result<ChatStream> {
          let request = ChatRequest {
              model: String::from("qwen3.5"),
              messages: messages.clone(),
              temperature,
          };

          let client = Client::new();
          let response = client
              .post(format!("{}/chat/completions", self.url))
              .json(&request)
              .send()
              .await
              .context("Failed to send request to server")?;

          if !response.status().is_success() {
              let error = response.text().await?;
              return Err(anyhow::anyhow!(error));
          }

          let stream = response
              .bytes_stream()
              .map_err(anyhow::Error::from)
              .map_err(|e| anyhow::anyhow!("{}", e))
              .collect::<FuturesBuffer<_>>()
              .await;

          Ok(stream)
      }
  }

  #[derive(Debug, Clone)]
  pub struct ChatStream {
      stream: futures::Stream<Box<dyn futures::Stream<Item = Result<Vec<u8>, anyhow::Error>
  + Send>>,
  }

  impl ChatStream {
      pub async fn receive(&mut self) -> Result<String> {
          let mut chunk = Vec::new();

          while let Some(chunk) = self.stream.next().await {
              let chunk = chunk?;
              chunk.extend(chunk);

              if chunk.contains(&b"\n") {
                  return Ok(String::from_utf8_lossy(&chunk).to_string());
              }
          }

          Ok(String::new())
      }
  }

              app: App {
                  messages: Vec::new(),
              },
          }
      }

      pub fn render(&mut self, frame: &mut Frame) {
          frame.render_widget(&self.app, frame.area());
      }

      pub fn handle_key(&mut self, event: Event) {
          if let Event::Key(key) = event {
              if key.code == KeyCode::Char('q') &&
  key.modifiers.contains(KeyModifiers::CONTROL) {
                  // Quit application
                  return;
              }
          }
      }
  }

  Key Features

  1. Async Chat Stream - Uses tokio and reqwest for streaming responses from llama.cpp
  2. File Operations - Basic read/write/edit commands
  3. Minimal TUI - Uses ratatui for terminal UI
  4. Error Handling - Comprehensive error handling with anyhow
  5. System Prompts - Supports custom system prompts for different AI behaviors
  6. Streaming Support - Uses SSE/Streaming API for efficient token generation
  7. Message Context - Maintains conversation history with role-based messages

  This minimal implementation provides a foundation for an AI coding assistant that can
  interact with llama.cpp server models and perform basic file operations through a terminal
   UI.

  Requirements

  1. llama.cpp server running on default port 8080
  2. Qwen3.5 model loaded on the server
  3. Network access to the llama.cpp server
  4. Terminal emulator that supports ratatui

  Error Handling

  The implementation includes robust error handling for:
  - Connection issues with llama.cpp server
  - JSON parsing errors from streaming responses
  - File operation failures
  - Invalid user input or commands
