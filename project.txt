  use agent::Agent;
  use clap::Parser;
  use std::path::PathBuf;

  #[derive(Parser, Debug)]
  #[command(author, version, about, long_about = None)]
  struct Args {
      /// Path to llama.cpp server socket or API endpoint
      #[arg(short, long)]
      server: PathBuf,

      /// Model name on the server
      #[arg(short, long, default_value = "qwen3.5")]
      model: String,

      /// Temperature for generation
      #[arg(short, long, default_value = "0.7")]
      temperature: f32,

      /// Maximum tokens
      #[arg(short, long, default_value = "2048")]
      max_tokens: u32,
  }

  #[tokio::main]
  async fn main() -> anyhow::Result<()> {
      let args = Args::parse();

      tracing::info!("Starting Qwen Agent TUI...");
      tracing::info!("Connecting to: {:?}", args.server);

      // Initialize agent
      let agent = Agent::new(
          args.server.to_string_lossy().to_string(),
          args.model.to_string(),
          args.temperature,
          args.max_tokens,
      )?;

      // Start TUI
      tui::run_tui(agent).await?;

      Ok(())
  }

  use std::sync::Arc;

  pub struct Agent {
      client: LlamaClient,
      tools: ToolRegistry,
      session: Arc<Mutex<SessionState>>,
  }

  #[derive(Default)]
  struct SessionState {
      history: Vec<ChatMessage>,
      current_file: Option<PathBuf>,
      current_command: String,
  }

  #[derive(Debug, Clone)]
  enum ChatMessage {
      User(String),
      Assistant(String),
      System(String),
  }

  impl Agent {
      pub fn new(
          server_url: String,
          model: String,
          temperature: f32,
          max_tokens: u32,
      ) -> anyhow::Result<Self> {
          let client = LlamaClient::new(server_url, model, temperature, max_tokens)?;
          let tools = ToolRegistry::default();
          let session = Arc::new(Mutex::new(SessionState::default()));

          Ok(Self {
              client,
              tools,
              session,
          })
      }

      pub async fn send_message(&self, message: &str) -> anyhow::Result<String> {
          let mut session = self.session.lock().await;
          session.history.push(ChatMessage::User(message.to_string()));

          let response = self.client.chat(message).await?;

          session.history.push(ChatMessage::Assistant(response.clone()));
          Ok(response)
      }

      pub async fn execute_command(&self, command: &str) -> anyhow::Result<String> {
          let mut session = self.session.lock().await;
          session.current_command = command.to_string();

          // Register available tools for the agent
          let tools = self.tools.get_tools();

          // Create a prompt with tool descriptions
          let tool_context = tools.as_json();
          let prompt = format!(
              "You are a coding assistant. Available tools:\n{}\nUser command: {}",
              tool_context, command
          );

          let response = self.client.chat(&prompt).await?;
          Ok(response)
      }

      pub fn get_history(&self) -> Vec<String> {
          let session = self.session.lock().unwrap();
          session.history
              .iter()
              .map(|m| match m {
                  ChatMessage::User(msg) => format!("User: {}", msg),
                  ChatMessage::Assistant(msg) => format!("Assistant: {}", msg),
                  ChatMessage::System(msg) => format!("System: {}", msg),
              })
              .collect()
      }

      pub fn get_file(&self) -> Option<PathBuf> {
          let session = self.session.lock().unwrap();
          session.current_file.clone()
      }

      pub fn set_file(&mut self, path: PathBuf) {
          let mut session = self.session.lock().unwrap();
          session.current_file = Some(path);
      }

      pub fn get_current_command(&self) -> String {
          let session = self.session.lock().unwrap();
          session.current_command.clone()
      }
  }

  4. src/file_ops.rs

  use std::fs::{self, File};
  use std::io::{BufRead, BufReader, Write};
  use std::path::PathBuf;
  use anyhow::{Context, Result};

  pub struct FileOperations {
      current_file: Option<PathBuf>,
  }

  impl FileOperations {
      pub fn new() -> Self {
          Self {
              current_file: None,
          }
      }

      pub fn open_file(&mut self, path: PathBuf) -> Result<String> {
          let content = fs::read_to_string(&path)?;
          self.current_file = Some(path);
          Ok(content)
      }

      pub fn read_file(&self, path: &PathBuf) -> Result<String> {
          let content = fs::read_to_string(path)
              .context(format!("Failed to read file: {}", path.display()))?;
          Ok(content)
      }

      pub fn write_file(&self, path: &PathBuf, content: &str) -> Result<String> {
          // Read existing content
          let existing = self.read_file(path)?;

          // Create modified content
          let new_content = Self::apply_changes(&existing, content);

          // Write to file
          fs::write(path, &new_content)?;

          // Truncate if we only want to write new content
          if content.is_empty() {
              fs::write(path, "")?;
          }

          Ok(new_content)
      }

      pub fn edit_file(
          &self,
          path: &PathBuf,
          operations: &str,
      ) -> Result<String> {
          let existing = self.read_file(path)?;
          let result = Self::apply_operations(&existing, operations)?;
          fs::write(path, &result)?;
          Ok(result)
      }

      pub fn list_directory(&self, path: &PathBuf) -> Result<Vec<PathBuf>> {
          let entries = fs::read_dir(path)?
              .filter_map(|e| e.ok())
              .map(|e| e.path())
              .collect();
          Ok(entries)
      }

      pub fn run_command(&self, cmd: &str) -> Result<String> {
          // Use shell command execution
          let output = std::process::Command::new("sh")
              .arg("-c")
              .arg(cmd)
              .output()?;

          let stdout = String::from_utf8_lossy(&output.stdout);
          let stderr = String::from_utf8_lossy(&output.stderr);

          if output.status.success() {
              Ok(stdout.to_string())
          } else {
              Err(anyhow::anyhow!(format!("Command failed: {}", stderr)))
          }
      }

      fn apply_changes(existing: &str, changes: &str) -> String {
          // Simple replacement logic
          if let Some(start) = changes.find("INSERT") {
              let insert_pos = existing.find("INSERT")
                  .or_else(|| existing.find("INSERT"))
                  .expect("No INSERT marker found");
              existing[..insert_pos + start].to_string()
                  + &existing[insert_pos..]
          } else if let Some(start) = changes.find("REPLACE") {
              let replace_range = existing.find("REPLACE")
                  .or_else(|| existing.find("REPLACE"))
                  .expect("No REPLACE marker found");
              existing[..replace_range].to_string()
                  + changes[4..].to_string().as_str()
          } else {
              // Default: just return the changes as new content
              changes.to_string()
          }
      }

      fn apply_operations(existing: &str, operations: &str) -> Result<String> {
          let lines = existing.lines();
          let mut new_content = Vec::new();

          for (i, line) in lines.enumerate() {
              // Check if this line needs to be modified
              if let Some(marker_pos) = operations.find(&line.to_string()) {
                  // Extract replacement from operations
                  let marker = operations[marker_pos..].split(',').next();
                  if let Some(marker) = marker {
                      let (prefix, replacement) = marker.split_once(':').unwrap_or((marker,
  ""));
                      new_content.push(if prefix == "INSERT" {
                          replacement.to_string()
                      } else if prefix == "REPLACE" {
                          replacement.to_string()
                      } else {
                          line.to_string()
                      })
                  } else {
                      new_content.push(line.to_string())
                  }
              } else {
                  new_content.push(line.to_string())
              }
          }

          Ok(new_content.join("\n"))
      }

      pub fn get_current_file(&self) -> Option<PathBuf> {
          self.current_file.clone()
      }

      pub fn get_current_content(&self) -> Option<String> {
          self.current_file.as_ref()
              .and_then(|path| self.read_file(path).ok())
      }
  }
